                    border: none;

                }
                .phone-input:focus {
                    background: #f8f9ff;
                }
                .input-hint {
                    display: block;
                    text-align: left;
                    margin-top: 8px;
                    color: #888;
                    font-size: 13px;
                }
            </style>
        `,
        showCancelButton: true,
        confirmButtonText: 'Proceed to Payment',
        confirmButtonColor: '#6e8efb',
        focusConfirm: false,
        preConfirm: () => {
            const phone = document.getElementById('payment-phone').value;
            if (!phone || !/^[0-9]{9}$/.test(phone)) {
                Swal.showValidationMessage('Please enter a valid 9-digit phone number');
                return false;
            }
            
            const formattedPhone = '254' + phone;
            return { phone: formattedPhone };
        }
    }).then((result) => {
        if (result.isConfirmed) {
            processPayment(id, name, price, currency, provider, result.value.phone, duration);
        }
    });
}

function processPayment(id, name, price, currency, provider, phone, duration) {
    Swal.fire({
        title: 'Processing Payment',
        html: `
            <div class="processing-payment">
                <div class="loading-spinner">
                    <div class="spinner"></div>
                </div>
                <p class="processing-text">Sending payment request to your ${provider === 'mpesa' ? 'MPESA' : 'Airtel Money'}...</p>
            </div>
            <style>
                .processing-payment {
                    text-align: center;
                }
                .loading-spinner {
                    margin: 20px auto;
                    width: 60px;
                    height: 60px;
                    position: relative;
                }
                .spinner {
                    width: 100%;
                    height: 100%;
                    border: 5px solid #f3f3f3;
                    border-top: 5px solid #6e8efb;
                    border-radius: 50%;
                    animation: spin 1s linear infinite;
                }
                @keyframes spin {
                    0% { transform: rotate(0deg); }
                    100% { transform: rotate(360deg); }
                }
                .processing-text {
                    color: #666;
                    margin-top: 15px;
                }
            </style>
        `,
        allowOutsideClick: false,
        showConfirmButton: false
    });
    
    // Prepare the payment data according to your controller requirements
    const paymentData = {
        product_id: id,
        phone: phone,
        user_id: escortwp_user_id,
        duration: duration,
        amount: price // Make sure to include the actual selected price
    };

    // Make the AJAX request to initiate payment
    jQuery.ajax({
        url: 'https://api.exoticnairobi.com/api/initiate-payment',
        method: 'POST',
        contentType: 'application/json',
        data: JSON.stringify(paymentData),
        success: function(response) {
            // Handle response based on your controller's return structure
            if (response.status === true) {
                Swal.fire({
                    title: 'Payment Request Sent!',
                    html: `
                        <div class="payment-success">
                            <div class="success-icon">📱</div>
                            <p>A payment request has been sent to your ${provider === 'mpesa' ? 'MPESA' : 'Airtel Money'} number.</p>
                            <p>Please complete the payment on your phone to activate your account.</p>
                            
                            <div class="manual-instructions">
                                <h4>If you don't receive the prompt:</h4>
                                <ol>
                                    <li>Go to your ${provider === 'mpesa' ? 'MPESA' : 'Airtel Money'} menu</li>
                                    <li>Select "Pay Bill"</li>
                                    <li>Enter Business Number <strong>68</strong></li>
                                    <li>Enter Account Number <strong>${phone}</strong></li>
                                    <li>Enter Amount <strong>${price}</strong></li>
                                </ol>
                            </div>
                        </div>
                        <style>
                            .payment-success {
                                text-align: center;
                            }
                            .success-icon {
                                font-size: 50px;
                                margin-bottom: 15px;
                            }
                            .manual-instructions {
                                background: #f8f9ff;
                                padding: 15px;
                                border-radius: 10px;
                                margin-top: 20px;
                                text-align: left;
                            }
                            .manual-instructions h4 {
                                margin-top: 0;
                                color: #333;
                            }
                            .manual-instructions ol {
                                padding-left: 20px;
                                margin-bottom: 0;
                            }
                            .manual-instructions li {
                                margin-bottom: 8px;
                            }
                        </style>
                    `,
                    icon: 'success',
                    confirmButtonColor: '#6e8efb'
                });
            } else {
                // Handle error cases from your API
                const errorMsg = response.error || 
                                response.message || 
                                'Payment initiation failed. Please try again.';
                
                Swal.fire({
                    title: 'Payment Failed',
                    text: errorMsg,
                    icon: 'error',
                    confirmButtonColor: '#6e8efb'
                });
            }
        },
        error: function(xhr) {
            // Handle AJAX errors
            let errorMessage = 'Payment initiation failed. Please try again.';
            
            try {
                const response = xhr.responseJSON;
                errorMessage = response.error || 
                              response.message || 
                              errorMessage;
            } catch (e) {
                console.error('Error parsing error response:', e);
            }
            
            Swal.fire({
                title: 'Error',
                text: errorMessage,
                icon: 'error',
                confirmButtonColor: '#6e8efb'
            });
        }
    });
}
